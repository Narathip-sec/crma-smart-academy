# CRMA Web — Build Skill

## Mission
Ship a LIFF web replacement for the Expo/RN app in `app/` with feature parity for cadets at CRMA.
Single codebase forward; mobile `app/` is frozen reference.

## Hard rules (never violate)
1. @crma.ac.th email gate enforced server-side at /api/auth/email-otp. Reject all other domains.
2. TOTP 2FA enrolled on first login, verified on every new device fingerprint.
3. Current-semester grades LOCKED at the Prisma query layer (Semester.isLocked).
4. Duty roster mutations RBAC-gated — Cadet read-only, Officer+ write.
5. Every API mutation writes an AuditLog row (userId, action, resource, ip, ua).
6. PDPA: encrypt User.totpSecret, User.email, PFTResult.*, Enrollment.grade with pgcrypto.
7. Caveman comms in PR descriptions, commit bodies, and chat. Code/security/commits = normal prose.
8. Bilingual README parity: every README.md edit mirrors to README_TH.md.
9. tsc --noEmit strict + noUncheckedIndexedAccess. All tests green = DoD per phase.
10. Superpowers loop per feature: brainstorm → plan → TDD → review → commit.

## Stack (locked)
- Next.js 15 (App Router, Node runtime for Prisma)
- TypeScript 5.4 strict
- Tailwind v4 + class-variance-authority
- Zustand 5 (slices: tab, schedule, health, activity)
- Prisma 5 → PostgreSQL (Supabase, region ap-southeast-1 SG)
- LIFF SDK v2.x (web-only mode acceptable for dev)
- next-intl (TH default, EN fallback)
- jose for JWT (httpOnly Secure cookies, SameSite=Lax)
- otplib for TOTP, Brevo SMTP for email OTP
- Vercel Cron + Inngest for jobs · Vercel Blob for files
- Vitest + @testing-library/react + Playwright + next-test-api-route

## Routing contract
Preserve mobile-app pattern: single activeTab string drives a switch.
Web mirrors via ?tab=home|class_schedule|health|activity|service|grades|me backed by
Zustand useTabStore. No nested route layouts. AdaptiveNav mutates store, store syncs URL.

## Definition of done per phase
- All new components have testIDs matching mobile (service.link.${id}, …).
- API routes enforce RBAC + write AuditLog.
- i18n keys exist in BOTH messages/th.json and messages/en.json.
- Playwright run inside LINE webview UA passes.
- Lighthouse perf budget: LCP < 2.5s on 4G throttling, CLS < 0.1.

## Forbidden
- react-native-web shim. Manual ports only.
- Skipping audit log "for a quick test" — never.
- Storing tokens in localStorage (LINE webview evicts it).
- Adding US-only managed services for primary data (sovereignty).

## When in doubt
- Read EXPORT.html and MIGRATION_LIFF.md at repo root.
- Source of truth for visuals: prototype_spec.md + references/*.png.
- Source of truth for data: app/src/fixtures/*.ts.

5 · Auth Logic (LINE → email OTP → TOTP)

[client] [server] liff.init() liff.login() if !logged in liff.getIDToken() ─────────────────► POST /api/auth/line/callback 1. verify LINE ID token (JWKs) 2. lookup User by lineUserId ├─ exists + verified → mint JWT └─ new → return 202 {next: "email"} POST /api/auth/email-otp/start ────► generate 6-digit OTP email to <input>@crma.ac.th REJECT if !endsWith("@crma.ac.th") POST /api/auth/email-otp/verify ───► validate OTP → set emailVerified bind lineUserId return 202 {next: "totp"} POST /api/auth/totp/enrol ─────► generate secret (otplib) return otpauth:// URL → QR POST /api/auth/totp/verify ─────► verify TOTP → encrypt+store secret mint JWT (httpOnly, Secure, SameSite=Lax) set-cookie: session=<jwt>
// lib/session.ts — JWT helpers (jose)
import { SignJWT, jwtVerify } from "jose";

const secret = new TextEncoder().encode(process.env.JWT_SECRET!);

export async function sign(userId: string, role: string) {
  return await new SignJWT({ sub: userId, role })
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setExpirationTime("1h")
    .sign(secret);
}

export async function verify(token: string) {
  const { payload } = await jwtVerify(token, secret);
  return payload as { sub: string; role: string };
}
// app/api/auth/email-otp/start/route.ts
import { NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";
import { sendOtpEmail } from "@/lib/email";

const CRMA_DOMAIN = /^[a-z0-9._%+-]+@crma\.ac\.th$/i;

export async function POST(req: Request) {
  const { email } = await req.json();
  if (!CRMA_DOMAIN.test(email)) {
    return NextResponse.json({ error: "domain_not_allowed" }, { status: 403 });
  }
  const otp = String(Math.floor(100000 + Math.random() * 900000));
  await prisma.emailOtp.create({ data: { email, otpHash: hash(otp),
    expiresAt: new Date(Date.now() + 10 * 60_000) }});
  await sendOtpEmail(email, otp);
  return NextResponse.json({ ok: true });
}
// middleware.ts — session gate + audit
import { NextResponse, type NextRequest } from "next/server";
import { verify } from "@/lib/session";

export async function middleware(req: NextRequest) {
  const isPublic = req.nextUrl.pathname.startsWith("/login")
    || req.nextUrl.pathname.startsWith("/api/auth");
  if (isPublic) return NextResponse.next();

  const token = req.cookies.get("session")?.value;
  if (!token) return NextResponse.redirect(new URL("/login", req.url));
  try { await verify(token); } catch {
    return NextResponse.redirect(new URL("/login", req.url));
  }
  return NextResponse.next();
}

"use client";
import { Home, CalendarDays, Heart, Users, LayoutGrid } from "lucide-react";
import { useTabStore } from "@/store/useTabStore";
import { useRouter, useSearchParams } from "next/navigation";
import { useEffect } from "react";
import clsx from "clsx";

const TABS = [
  { key: "home",           icon: Home,         label: "nav.home" },
  { key: "class_schedule", icon: CalendarDays, label: "nav.class" },
  { key: "health",         icon: Heart,        label: "nav.health" },
  { key: "activity",       icon: Users,        label: "nav.activity" },
  { key: "service",        icon: LayoutGrid,   label: "nav.service" },
] as const;

export function BottomNav() {
  const { activeTab, setActiveTab } = useTabStore();
  const router = useRouter();
  const sp = useSearchParams();

  // keep URL in sync with store (back/forward + deep-link)
  useEffect(() => {
    const url = new URLSearchParams(sp);
    if (url.get("tab") !== activeTab) {
      url.set("tab", activeTab);
      router.replace(`/?${url.toString()}`, { scroll: false });
    }
  }, [activeTab]);

  return (
    <nav className="absolute bottom-0 inset-x-0 z-40 bg-white rounded-t-[2.5rem]
      flex justify-around px-2 py-3 shadow-[0_-8px_30px_rgb(0,0,0,0.04)]">
      {TABS.map(({ key, icon: Icon, label }) => {
        const active = activeTab === key;
        return (
          <button key={key} data-testid={`bottomnav.${key}`}
            onClick={() => setActiveTab(key)}
            className={clsx("flex flex-col items-center gap-1 rounded-2xl px-3 py-2 transition-transform",
              active && "bg-amber-400 scale-110 text-slate-900",
              !active && "text-slate-500")}>
            <Icon size={20} strokeWidth={active ? 2.4 : 2} />
          </button>
        );
      })}
    </nav>
  );
}

6.4 · store/useTabStore.ts — Zustand + URL sync (new) 
"use client";
import { create } from "zustand";

export type TabKey =
  | "home" | "class_schedule" | "health" | "activity"
  | "service" | "grades" | "me";

interface TabState {
  activeTab: TabKey;
  setActiveTab: (t: TabKey) => void;
}

export const useTabStore = create<TabState>((set) => ({
  activeTab: "home",
  setActiveTab: (activeTab) => set({ activeTab }),
}));

// Reads URL on mount and pushes to store. Mount once at top of (app)/layout.
export function useTabUrlSync() {
  const setActiveTab = useTabStore((s) => s.setActiveTab);
  const sp = useSearchParams();
  useEffect(() => {
    const raw = sp.get("tab") as TabKey | null;
    if (raw) setActiveTab(raw);
  }, [sp]);
}

7 · API Logic Pattern (RBAC + Prisma + Audit)

// app/api/grades/route.ts — exemplar. Mirror this shape across all routes.
import { NextResponse } from "next/server";
import { requireSession } from "@/lib/session";
import { canViewGrades } from "@/lib/rbac";
import { audit } from "@/lib/audit";
import { prisma } from "@/lib/prisma";

export async function GET(req: Request) {
  const session = await requireSession(req);
  if (!canViewGrades(session.user))
    return new NextResponse(null, { status: 403 });

  const code = new URL(req.url).searchParams.get("semester");
  const sem = await prisma.semester.findUnique({ where: { code: code! }});
  if (!sem)        return NextResponse.json({ error: "not_found" }, { status: 404 });
  if (sem.isLocked) return NextResponse.json({ error: "locked" }, { status: 423 });

  const rows = await prisma.enrollment.findMany({
    where: { userId: session.user.id, course: { semesterId: sem.id }},
    include: { course: true },
  });
  await audit(session.user.id, "READ:grades", `semester:${sem.code}`, req);
  return NextResponse.json(rows);
}
// lib/rbac.ts — port + extend from app/src/lib/rbac.ts
import type { Role } from "@prisma/client";

export const canEditRoster   = (u: { role: Role }) =>
  u.role === "OFFICER" || u.role === "ADMIN";
export const canViewGrades   = (u: { role: Role }) => true;
export const canRecordPft    = (u: { role: Role }) =>
  u.role === "INSTRUCTOR" || u.role === "ADMIN";
export const canPushAnnouncement = (u: { role: Role }) =>
  u.role === "OFFICER" || u.role === "ADMIN" || u.role === "INSTRUCTOR";
// lib/audit.ts
import { prisma } from "@/lib/prisma";

export async function audit(
  userId: string | null,
  action: string,
  resource: string,
  req: Request,
) {
  await prisma.auditLog.create({
    data: { userId, action, resource,
      ip: req.headers.get("x-forwarded-for"),
      userAgent: req.headers.get("user-agent") }
  });
}
